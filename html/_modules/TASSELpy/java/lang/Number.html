    __metaclass__ = ABCMeta
    @classmethod
    def __subclasshook__(cls, C):
        if issubclass(C,np.number):
            return True
        elif any(map(lambda x: issubclass(C,x),(int, long, float))):
            return True
        elif issubclass(C, Number):
            return True
        else:
            return False
